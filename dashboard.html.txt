<!DOCTYPE html>
<html>
<head>
  <title>Dashboard</title>

  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getFirestore, collection, addDoc, getDocs } 
    from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

    const firebaseConfig = {
      apiKey: "YOUR_CONFIG",
      authDomain: "YOUR_CONFIG",
      projectId: "YOUR_CONFIG",
    };

    const app = initializeApp(firebaseConfig);
    const db = getFirestore(app);

    window.saveScore = async function(score){
      await addDoc(collection(db, "scores"), {
        marks: score,
        date: new Date().toLocaleString()
      });
      loadScores();
    }

    window.loadScores = async function(){
      let data = await getDocs(collection(db, "scores"));
      let html = "<h3>Report Card</h3>";

      data.forEach(doc=>{
        html += `<p>Score: ${doc.data().marks}</p>`;
      });

      document.getElementById("report").innerHTML = html;
    }

    window.onload = loadScores;
  </script>

  <style>
    body {background:#0b0f19; color:white; text-align:center;}
    button {padding:10px; margin:20px; background:#3b82f6;}
  </style>
</head>
